  function Recurse(Node : PLogOper) : Boolean;

  var

    O1, O2 : Boolean;

  begin

    If Node^.i1 = nil then O1 := GetInput(Node^.C1) else

      O1 := Recurse(Node^.i1);

        O2 := Recurse(Node^.i2);

      loAnd : Result := O1 and O2; // •

      loNot : Result := not O1;    // ¬

      else Result := O1;
